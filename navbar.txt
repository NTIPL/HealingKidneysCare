CustomNavbar.js

import React, { useState } from 'react';
import { Navbar, Nav, Container, Row, Col } from 'react-bootstrap'; // Importing Bootstrap components
import { PhoneOutlined, MailOutlined } from '@ant-design/icons';
import FaxIcon from './InputElements/FaxIcon'; // Ensure correct path to FaxIcon
import logo from '../assets/HKI-2019-Logo.png.webp'; // Ensure correct path to logo
import './CustomNavbar.css'; // Adjust the path if necessary

const CustomNavbar = () => {
    const [current, setCurrent] = useState('1');

    // List of menu items
    const menuList = [
        'Home',
        'Physicians',
        'Services',
        'Patient Registration',
        'Education',
        'Locations/Directions',
    ];

    // Handle menu item click
    const handleClick = (key) => {
        setCurrent(key);
    };

    return (
        <div>
            {/* Contact Info Section */}
            <div className="bg-light py-3 border-b border-gray-200"> {/* Increased padding */}
                <Container>
                    <Row className="justify-content-center align-items-center">
                        <Col xs={12} className="text-center"> {/* Center the text */}
                            <div className="flex justify-center flex-wrap"> {/* Use flex-wrap for mobile responsiveness */}
                                <div className="flex items-center border border-gray-300 rounded-lg bg-lightblue-200 mx-2 p-2 contact-info"> {/* Rounded and light blue */}
                                    <PhoneOutlined className="me-2" />
                                    (832) 610-2822
                                </div>
                                <div className="flex items-center border border-gray-300 rounded-lg bg-lightblue-200 mx-2 p-2 contact-info"> {/* Rounded and light blue */}
                                    <FaxIcon className="me-2" />
                                    (936) 777-8831
                                </div>
                                <div className="flex items-center border border-gray-300 rounded-lg bg-lightblue-200 mx-2 p-2 contact-info"> {/* Rounded and light blue */}
                                    <MailOutlined className="me-2" />
                                    healingkidneys@gmail.com
                                </div>
                            </div>
                        </Col>
                    </Row>
                </Container>
            </div>

            {/* Navbar */}
            <Navbar expand="lg" className="bg-pink-200"> {/* Light pink background */}
                <Container>
                    {/* Logo on the left side */}
                    <Navbar.Brand href="#">
                        <img src={logo} alt="Logo" style={{ width: '170px' }} />
                    </Navbar.Brand>

                    {/* Hamburger for mobile view */}
                    <Navbar.Toggle aria-controls="navbar-nav" />

                    {/* Navbar links */}
                    <Navbar.Collapse id="navbar-nav">
                        <Nav className="ms-auto">
                            {menuList.map((item, index) => (
                                <Nav.Link
                                    key={index}
                                    href={`#${item.toLowerCase().replace(/ /g, '-')}`}
                                    onClick={() => handleClick((index + 1).toString())}
                                    active={current === (index + 1).toString()}
                                    className="px-4 nav-link-custom" 
                                >
                                    {item}
                                </Nav.Link>
                            ))}
                        </Nav>
                    </Navbar.Collapse>
                </Container>
            </Navbar>
        </div>
    );
};

export default CustomNavbar;
